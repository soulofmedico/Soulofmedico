NEW CODE IN lectures.js:

function openVideo(lectureId) {
    // 1. Find lecture in database
    currentLecture = lecturesData.find(l => l.id === lectureId);
    
    // 2. Validate lecture exists
    if (!currentLecture) {
        console.error('❌ ERROR: Lecture not found with ID:', lectureId);
        alert('❌ Lecture not found. Please refresh and try again.');
        return;
    }

    // 3. Validate YouTube ID
    if (!currentLecture.youtubeId || currentLecture.youtubeId.trim() === '') {
        console.error('❌ ERROR: No YouTube ID found');
        alert('❌ Video ID missing. Please check data.js');
        return;
    }

    // 4. Show video player
    document.getElementById('lectureListView').style.display = 'none';
    document.getElementById('videoPlayerView').style.display = 'flex';

    // 5. Build YouTube URL with parameters
    const youtubeUrl = `https://www.youtube.com/embed/${currentLecture.youtubeId}
        ?autoplay=0&controls=1&modestbranding=1&rel=0&fs=1&playsinline=1`;

    // 6. Embed iframe with all permissions
    const videoContainer = document.getElementById('videoContainer');
    videoContainer.innerHTML = `
        <iframe 
            class="lecture-video"
            src="${youtubeUrl}"
            width="100%"
            height="100%"
            frameborder="0"
            allowfullscreen
            allow="accelerometer; autoplay; clipboard-write; encrypted-media; 
                   gyroscope; picture-in-picture; web-share"
            loading="lazy"
            title="${currentLecture.title}">
        </iframe>
    `;

    // 7. Show error if something fails
    try {
        // ... embed code
    } catch (error) {
        videoContainer.innerHTML = `
            <div style="...error styling...">
                ❌ Video Failed to Load
                YouTube ID: ${currentLecture.youtubeId}
                Check: 1. ID is correct 2. Video is public 3. Refresh
            </div>
        `;
    }
}

GETTING YOUTUBE VIDEO ID:

YouTube URL:  https://www.youtube.com/watch?v=58GV7_WTCoKzaX_Y
                                                ^^^^^^^^^^^^^^ This is the ID

ID Format:    • Exactly 11 characters
              • Mix of letters, numbers, hyphens, underscores
              • No spaces
              • Case-sensitive

CORRECT ✅:
  youtubeId: "58GV7_WTCoKzaX_Y"
  youtubeId: "dQw4w9WgXcQ"
  youtubeId: "jNQXAC9IVRw"

WRONG ❌:
  youtubeId: "https://www.youtube.com/watch?v=58GV7_WTCoKzaX_Y"  (full URL)
  youtubeId: "58GV7_WTCoKzaX_Y "  (extra space)
